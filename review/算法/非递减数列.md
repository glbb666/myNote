var checkPossibility = function(nums) {
    if(nums.length<=2){
        return true;
    }
    let flag = false;
    for(let index = 0;index<nums.length-1;index++){
        if(nums[index]>nums[index+1]){
            if(flag){
                return false;
            }
            if(index-1>=0){
                if(nums[index-1]>nums[index+1]){
                    nums[index+1] = nums[index];
                }else{
                    nums[index] = nums[index-1];
                }
            }else{
                nums[index] = nums[index+1];
            }
            flag = true; 
        }
    }
    return true;
};
