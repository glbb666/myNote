思路：利用归并排序。先分，分完之后在排序的过程中进行计数。让指针分别指向左数组和右数组的尾端，如果左指针指向的值的比右指针指向的值大，则说明

```javascript
/**
 * @param {number[]} nums
 * @return {number}
 */

var reversePairs = function(nums) {
    let count = 0;
    function reverParirsMerge(left,right){
        let il = left.length-1;
        let ir = right.length-1;
        const result = [];
        while(il>=0&&ir>=0){
            if(left[il]>right[ir]){
                count+=(ir+1);
                result.unshift(left[il--]);
            }else{
                result.unshift(right[ir--]);
            }
        }
        while(il>=0){ 
            result.unshift(left[il--]);
        }
        while(ir>=0){
            result.unshift(right[ir--]);
        }
        return result;
    };
    function reversePairsSort(nums){
        if(nums.length<=1){
            return nums;
        }
        const mid = Math.floor(nums.length/2);
        const left = nums.slice(0,mid);
        const right = nums.slice(mid);
        return reverParirsMerge(reversePairsSort(left),reversePairsSort(right));
    }
    reversePairsSort(nums);
    return count;
};
```
